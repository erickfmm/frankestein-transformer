 34        """Initialize the quantizer with default BitNet b1.58 config."""
 35        self.quantization_config = {
 36            'weight_bits': 1.58,  # Ternary: {-1, 0, 1}
 37            'activation_bits': 8,   # INT8 for activations
 38        }

 40    @staticmethod
 41    def quantize_ternary_weights(weight: torch.Tensor) -> Tuple[np.ndarray, float]:
 42        """
 43        Quantize weights to ternary {-1, 0, 1} and pack for storage.
 44        
 45        Args:
 46            weight: Float tensor to quantize
 47            
 48        Returns:
 49            packed_weights: Packed ternary weights (2 bits per weight, stored efficiently)
 50            scale: Scaling factor for dequantization
 51        """
 52        # Calculate scale
 53        scale = weight.abs().mean().item()
 54        if scale < 1e-5:
 55            scale = 1.0
 56            
 57        # Quantize to {-1, 0, 1}
 58        w_scaled = weight / scale
 59        w_ternary = torch.round(w_scaled).clamp(-1, 1).cpu().numpy()
 60        
 61        # Pack ternary values: -1 -> 0b00, 0 -> 0b01, 1 -> 0b10
 62        # This allows 4 weights per byte (2 bits each)
 63        packed = BitNetQuantizer._pack_ternary(w_ternary.flatten())
 64        
 65        return packed, scale
 66    
 67    @staticmethod
 68    def _pack_ternary(ternary_array: np.ndarray) -> np.ndarray:
 69        """
 70        Pack ternary values into 2 bits each (4 values per byte).
 71        Mapping: -1 -> 0, 0 -> 1, 1 -> 2
 72        """
 73        # Convert -1, 0, 1 to 0, 1, 2
 74        packed_values = (ternary_array + 1).astype(np.uint8)
 75        
 76        # Pack 4 values into each byte
 77        n = len(packed_values)
 78        n_bytes = (n + 3) // 4  # Round up
 79        
 80        packed = np.zeros(n_bytes, dtype=np.uint8)
 81        for i in range(n):
 82            byte_idx = i // 4
 83            bit_offset = (i % 4) * 2
 84            packed[byte_idx] |= (packed_values[i] & 0b11) << bit_offset
 85            
 86        return packed
 87    
 88    @staticmethod
 89    def _unpack_ternary(packed: np.ndarray, original_size: int) -> np.ndarray:
 90        """
 91        Unpack ternary values from 2-bit packed format.
 92        """
 93        unpacked = np.zeros(original_size, dtype=np.int8)
 94        
 95        for i in range(original_size):
 96            byte_idx = i // 4
 97            bit_offset = (i % 4) * 2
 98            value = (packed[byte_idx] >> bit_offset) & 0b11
 99            unpacked[i] = value - 1  # Convert back to -1, 0, 1
100            
101        return unpacked

103    @staticmethod
104    def dequantize_ternary_weights(
105        packed: np.ndarray,
106        scale: float,
107        original_shape: Tuple[int, ...]
108    ) -> torch.Tensor:
109        """
110        Dequantize packed ternary weights back to float tensor.
111        
112        Args:
113            packed: Packed ternary weights
114            scale: Scaling factor
115            original_shape: Original tensor shape
116            
117        Returns:
118            Dequantized float tensor
119        """
120        n_elements = np.prod(original_shape)
121        ternary = BitNetQuantizer._unpack_ternary(packed, n_elements)
122        weights = torch.from_numpy(ternary).float() * scale
123        return weights.reshape(original_shape)

125    def quantize_model_weights(self, model: nn.Module) -> Dict[str, Any]:
126        """
127        Quantize all BitLinear weights in the model to ternary format.
128        
129        Args:
130            model: PyTorch model to quantize
131            
132        Returns:
133            Dictionary with quantized weights and metadata
134        """
135        quantized_state = {
136            'weights': {},
137            'scales': {},
138            'shapes': {},
139            'config': self.quantization_config
140        }
141        
142        total_original_size = 0
143        total_compressed_size = 0
144        
145        for name, param in model.named_parameters():
146            if 'weight' in name:
147                # Quantize weight
148                packed, scale = self.quantize_ternary_weights(param.data)
149                
150                quantized_state['weights'][name] = packed
151                quantized_state['scales'][name] = scale
152                quantized_state['shapes'][name] = tuple(param.shape)
153                
154                # Track compression
155                original_size = param.numel() * 4  # FP32 = 4 bytes
156                compressed_size = len(packed)
157                
158                total_original_size += original_size
159                total_compressed_size += compressed_size
160                
161                logger.debug(f"{name}: {original_size / 1024:.2f}KB -> {compressed_size / 1024:.2f}KB")
162            else:
163                # Keep biases and other parameters as-is (if any)
164                quantized_state['weights'][name] = param.data.cpu().numpy()
165        
166        compression_ratio = total_original_size / total_compressed_size
167        logger.info(f"Model compressed: {total_original_size / (1024**2):.2f}MB -> "
168                   f"{total_compressed_size / (1024**2):.2f}MB "
169                   f"(Compression ratio: {compression_ratio:.2f}x)")
170        
171        return quantized_state

174        self,
175        quantized_state: Dict[str, Any],
176        model: nn.Module
177    ) -> None:
178        """
179        Load quantized weights back into model.
180        
181        Args:
182            quantized_state: Dictionary with quantized weights
183            model: PyTorch model to load weights into
184        """
185        state_dict = {}
186        
187        for name in quantized_state['weights'].keys():

193                
194                weight = self.dequantize_ternary_weights(packed, scale, shape)
195                state_dict[name] = weight
196            else:
197                # Load non-quantized parameter
198                state_dict[name] = torch.from_numpy(quantized_state['weights'][name])
199        
200        model.load_state_dict(state_dict, strict=False)
201        logger.info(f"Loaded {len(state_dict)} quantized parameters into model")

204class ActivationQuantizer:
205    """Runtime activation quantization for inference optimization.
206
207    Provides static methods for quantizing activations to INT8 and
208    dequantizing back to float, enabling efficient integer-arithmetic
209    computation during inference.
210    """

212    @staticmethod
213    def quantize_activation_int8(x: torch.Tensor) -> Tuple[torch.Tensor, float]:
214        """Quantize activations to INT8 range ``[-128, 127]``.
215
216        Args:
217            x: Activation tensor of any shape.
218
219        Returns:
220            Tuple of ``(quantized_tensor, scale)`` where ``scale`` is the
221            factor used for quantization (for later dequantization).
222        """
223        scale = 127.0 / x.abs().max().clamp(min=1e-5)
224        x_q = (x * scale).round().clamp(-128, 127)
225        return x_q, scale.item()
226    
227    @staticmethod
228    def dequantize_activation_int8(
229        x_q: torch.Tensor,
230        scale: float
231    ) -> torch.Tensor:
232        """Dequantize INT8 activations back to float.
233
234        Args:
235            x_q: Quantized INT8 tensor.
236            scale: The scale factor from :meth:`quantize_activation_int8`.
237
238        Returns:
239            Dequantized float tensor.
240        """
241        return x_q.float() / scale

 54    """
 55
 56    def __init__(self, config: UltraConfig, device: str = "cpu"):
 57        """Initialize the deployer.
 58
 59        Args:
 60            config: Model configuration.
 61            device: PyTorch device string for conversion.
 62        """
 63        self.config = config
 64        self.device = device
 65        self.model = None
 66        
 67    def load_training_checkpoint(self, checkpoint_path: str) -> None:
 68        """
 69        Load model from training checkpoint.
 70        
 71        Args:
 72            checkpoint_path: Path to training checkpoint
 73        """
 74        logger.info(f"Loading training checkpoint: {checkpoint_path}")
 75        
 76        checkpoint = torch.load(checkpoint_path, map_location=self.device)
 77        
 78        # Extract config if saved with checkpoint
 79        if 'config' in checkpoint:
 80            saved_config = checkpoint['config']
 81            # Update config with saved values
 82            for key, value in saved_config.items():
 83                if hasattr(self.config, key):
 84                    setattr(self.config, key, value)
 85            logger.info("Loaded config from checkpoint")
 86        
 87        # Initialize model
 88        self.model = TormentedBertFrankenstein(self.config)
 89        self.model.to(self.device)
 90        
 91        # Load state dict
 92        if 'model_state_dict' in checkpoint:
 93            self.model.load_state_dict(checkpoint['model_state_dict'])
 94        elif 'state_dict' in checkpoint:
 95            self.model.load_state_dict(checkpoint['state_dict'])
 96        else:
 97            # Assume checkpoint is state dict
 98            self.model.load_state_dict(checkpoint)
 99        
100        logger.info("Model loaded successfully")
101        
102        # Print model info
103        total_params = sum(p.numel() for p in self.model.parameters())
104        logger.info(f"Model parameters: {total_params / 1e6:.2f}M")
105
