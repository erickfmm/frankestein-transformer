138        self,
139        sentence1: Union[str, List[str]],
140        sentence2: Union[str, List[str]],
141        metric: str = "cosine"
142    ) -> Union[float, np.ndarray]:
143        """
144        Compute similarity between sentence(s).
145        
146        Args:
147            sentence1: First sentence(s)
148            sentence2: Second sentence(s)
149            metric: Similarity metric ('cosine' or 'dot')
150

152            Similarity score(s) in range [0, 1] (or [-1, 1] for unnormalized)

154        # Encode sentences

157        
158        # Compute similarity

160            # For normalized embeddings, dot product = cosine similarity
161            similarity = np.sum(emb1 * emb2, axis=1) if emb1.ndim > 1 else np.dot(emb1, emb2)
162        elif metric == "dot":
163            similarity = np.sum(emb1 * emb2, axis=1) if emb1.ndim > 1 else np.dot(emb1, emb2)
164        else:
165            raise ValueError(f"Unknown metric: {metric}")
166        
167        return similarity

 563    def _parse_smi_value(self, raw: str) -> float:
 564        value = (raw or "").strip()
 565        if not value or value.lower() in {"n/a", "[not supported]"}:
 566            return 0.0
 567        match = re.search(r"[-+]?\d*\.?\d+", value)
 568        if match is None:
 569            return 0.0
 570        try:
 571            return float(match.group(0))
 572        except ValueError:
 573            return 0.0

 611    def _get_gpu_telemetry(self) -> Dict[str, float]:
 612        telemetry = self._get_default_gpu_telemetry()
 613        if not torch.cuda.is_available():
 614            return telemetry
 615        backend = self.gpu_metrics_backend.strip().lower()
 616        if backend == "none":
 617            return telemetry
 618        if backend != "nvml":
 619            return telemetry
 620
 621        if self._nvml_disabled:
 622            return self._get_gpu_telemetry_from_nvidia_smi()
 623
 624        if self._nvml_module is None or self._nvml_handle is None:
 625            try:
 626                import pynvml  # type: ignore
 627
 628                pynvml.nvmlInit()
 629                idx = int(self.gpu_temp_guard._nvml_device_index)
 630                self._nvml_handle = pynvml.nvmlDeviceGetHandleByIndex(idx)
 631                self._nvml_module = pynvml
 632            except Exception as exc:
 633                self._nvml_disabled = True
 634                if not self._nvml_warning_logged:
 635                    logger.warning("NVML telemetry unavailable, falling back to nvidia-smi: %s", exc)
 636                    self._nvml_warning_logged = True
 637                return self._get_gpu_telemetry_from_nvidia_smi()
 638
 639        try:
 640            pynvml = self._nvml_module
 641            handle = self._nvml_handle
 642            utilization = pynvml.nvmlDeviceGetUtilizationRates(handle)
 643            memory_info = pynvml.nvmlDeviceGetMemoryInfo(handle)
 644            telemetry["gpu_temp_c"] = float(
 645                pynvml.nvmlDeviceGetTemperature(handle, pynvml.NVML_TEMPERATURE_GPU)
 646            )
 647            telemetry["gpu_power_w"] = float(pynvml.nvmlDeviceGetPowerUsage(handle)) / 1000.0
 648            telemetry["gpu_util_pct"] = float(utilization.gpu)
 649            telemetry["gpu_mem_used_mib"] = float(memory_info.used) / (1024.0 ** 2)
 650        except Exception as exc:
 651            if not self._nvml_warning_logged:
 652                logger.warning("Failed reading NVML telemetry, falling back to nvidia-smi: %s", exc)
 653                self._nvml_warning_logged = True
 654            return self._get_gpu_telemetry_from_nvidia_smi()
 655        return telemetry

 769    def _prune_thermal_emergency_artifacts(self):
 770        max_keep = max(int(self.checkpoint_save_total_limit), 1)
 771        entries = []
 772        for entry in os.listdir(self.thermal_emergency_dir):
 773            full_path = os.path.join(self.thermal_emergency_dir, entry)
 774            try:
 775                mtime = os.path.getmtime(full_path)
 776            except OSError:
 777                continue
 778            m = re.search(r"(\d{8}_\d{6}_\d{6})$", entry)
 779            embedded_ts = m.group(1) if m else ""
 780            entries.append((embedded_ts, mtime, full_path))
 781
 782        entries.sort(key=lambda item: (item[0], item[1], item[2]))
 783        while len(entries) > max_keep:
 784            _, _, old_path = entries.pop(0)
 785            try:
 786                if os.path.isdir(old_path):
 787                    shutil.rmtree(old_path)
 788                elif os.path.exists(old_path):
 789                    os.remove(old_path)
 790                logger.info("Removed old SBERT thermal artifact: %s", old_path)
 791            except Exception as exc:
 792                logger.warning("Failed removing old SBERT thermal artifact %s: %s", old_path, exc)

 809            return False
 810        if self._force_cpu_only_after_gpu_error:
 811            return False

 815            return False

 817        logger.error(
 818            "[ThermalSwitch][SBERT] GPU runtime error detected; switching to CPU-only mode for remaining training: %s",

 838            torch.cuda.empty_cache()

 840        self._thermal_offload_started_monotonic = None
 841        self._thermal_last_poll_monotonic = 0.0

 845    def _offload_model_for_critical_thermal_event(self, batch_idx: int, temp_c: float) -> str:
 846        if not self._switch_on_thermal:
 847            return "none"
 848        if self._force_cpu_only_after_gpu_error:
 849            return "thermal_cpu_only_gpu_error"
 850        if self._thermal_offload_active:
 851            return self._monitor_thermal_offload_and_maybe_reload(batch_idx)
 852
 853        context = f"sbert.fit batch={batch_idx + 1}"
 854        logger.warning(
 855            "[ThermalOffload][SBERT] %s critical threshold reached at %.1fC; switching to CPU training mode",
 856            context,
 857            temp_c,
 858        )
 859
 860        model_snapshot_path: Optional[str] = None
 861        resume_artifact_path: Optional[str] = None
 862        try:
 863            model_snapshot_path = self._save_thermal_model_snapshot(batch_idx, temp_c)
 864        except Exception as exc:
 865            logger.exception(
 866                "Failed to save SBERT thermal model snapshot before offload: %s",
 867                exc,
 868            )
 869
 870        try:
 871            resume_artifact_path = self._save_thermal_resume_artifact_if_available(batch_idx)
 872        except Exception as exc:
 873            logger.exception(
 874                "Failed to save SBERT thermal resume artifact before offload: %s",
 875                exc,
 876            )
 877
 878        self._thermal_last_model_snapshot = model_snapshot_path
 879        self._thermal_last_resume_artifact = resume_artifact_path
 880        self._last_guard_temp_c = float(temp_c)
 881        self.model.to("cpu")
 882        if torch.cuda.is_available():
 883            torch.cuda.empty_cache()
 884        self._thermal_offload_active = True
 885        self._thermal_offload_started_monotonic = time.perf_counter()
 886        self._thermal_last_poll_monotonic = 0.0
 887        return "thermal_offload_cpu_mode"

 889    def _monitor_thermal_offload_and_maybe_reload(self, batch_idx: int) -> str:
 890        if not self._thermal_offload_active:
 891            return "none"
 892        if self._force_cpu_only_after_gpu_error:
 893            return "thermal_cpu_only_gpu_error"
 894
 895        now = time.perf_counter()
 896        started = (
 897            self._thermal_offload_started_monotonic
 898            if self._thermal_offload_started_monotonic is not None
 899            else now
 900        )
 901        elapsed_s = max(now - started, 0.0)
 902        poll_interval = float(self.gpu_temp_guard.poll_interval_seconds)
 903        if (now - self._thermal_last_poll_monotonic) < poll_interval:
 904            return f"thermal_cpu_training_{int(round(elapsed_s))}s"
 905
 906        self._thermal_last_poll_monotonic = now
 907        resume_threshold = float(self.gpu_temp_guard.resume_threshold_c)
 908        current_temp = float(self.gpu_temp_guard.read_temperature_c())
 909        self._last_guard_temp_c = current_temp
 910        context = f"sbert.fit batch={batch_idx + 1}"
 911        if current_temp > resume_threshold:
 912            logger.warning(
 913                "[ThermalOffload][SBERT] %s continuing CPU training: GPU %.1fC (resume <= %.1fC)",
 914                context,
 915                current_temp,
 916                resume_threshold,
 917            )
 918            return f"thermal_cpu_training_{int(round(elapsed_s))}s"
 919
 920        try:
 921            self.model.to(self.device)
 922        except Exception as exc:
 923            logger.error(
 924                "[ThermalSwitch][SBERT] Failed to reload to GPU after thermal CPU mode. "
 925                "Switching to CPU-only mode. model_snapshot=%s resume_artifact=%s error=%s",
 926                self._thermal_last_model_snapshot or "unavailable",
 927                self._thermal_last_resume_artifact or "unavailable",
 928                exc,
 929            )
 930            self._force_cpu_only_after_gpu_error = True
 931            self._thermal_offload_active = False
 932            self._thermal_offload_started_monotonic = None
 933            self._thermal_last_poll_monotonic = 0.0
 934            return "thermal_cpu_only_gpu_error"
 935
 936        self._thermal_offload_active = False
 937        self._thermal_offload_started_monotonic = None
 938        self._thermal_last_poll_monotonic = 0.0
 939        logger.warning(
 940            "[ThermalOffload][SBERT] %s resumed GPU training after %.1fs at %.1fC",
 941            context,
 942            elapsed_s,
 943            current_temp,
 944        )
 945        return f"thermal_onload_gpu_{int(round(elapsed_s))}s"

 947    def _handle_thermal_guard_for_batch(self, batch_idx: int) -> str:
 948        if not self.gpu_temp_guard.is_active:
 949            return "none"
 950        if self._force_cpu_only_after_gpu_error:
 951            return "thermal_cpu_only_gpu_error"
 952        if not self._switch_on_thermal:
 953            context = f"sbert.fit batch={batch_idx + 1}"
 954            temp_c = float(self.gpu_temp_guard.read_temperature_c())
 955            self._last_guard_temp_c = temp_c
 956            if temp_c > float(self.gpu_temp_guard.pause_threshold_c):
 957                result = self.gpu_temp_guard.wait_until_safe(context=context)
 958                self._last_guard_temp_c = float(result.temp_c)
 959                return result.repair_action
 960            return "none"
 961        if self._thermal_offload_active:
 962            return self._monitor_thermal_offload_and_maybe_reload(batch_idx)
 963
 964        context = f"sbert.fit batch={batch_idx + 1}"
 965        temp_c = float(self.gpu_temp_guard.read_temperature_c())
 966        self._last_guard_temp_c = temp_c
 967        critical_threshold = self.gpu_temp_guard.critical_threshold_c
 968        if critical_threshold is not None and temp_c >= float(critical_threshold):
 969            return self._offload_model_for_critical_thermal_event(batch_idx, temp_c)
 970        if temp_c > float(self.gpu_temp_guard.pause_threshold_c):
 971            result = self.gpu_temp_guard.wait_until_safe(context=context)
 972            self._last_guard_temp_c = float(result.temp_c)
 973            return result.repair_action
 974        return "none"

1002    def load_dataset(
1003        self,
1004        dataset_name: str = "erickfmm/agentlans__multilingual-sentences__paired_10_sts",
1005        dataset_type: str = "paired_similarity",
1006        columns: Optional[Dict[str, str]] = None,
1007        query_prefix: str = "",
1008        document_prefix: str = "",
1009        max_train_samples: Optional[int] = None,
1010        max_eval_samples: Optional[int] = 10000,
1011        resample_balanced: bool = True,
1012        standardize_scores: bool = False,
1013        target_std: float = 0.3
1014    ):
1015        """
1016        Load and prepare the STS dataset.
1017        
1018        Scores are kept in [-1, 1] range to preserve sign:
1019        - Negative scores remain negative (dissimilar sentences)
1020        - Zero remains zero (neutral similarity)
1021        - Positive scores remain positive (similar sentences)
1022        
1023        Args:
1024            dataset_name: HuggingFace dataset name
1025            dataset_type: Dataset format type (paired_similarity, triplets, qa)
1026            columns: Optional dataset column mapping
1027            query_prefix: Optional prefix prepended to query/question text
1028            document_prefix: Optional prefix prepended to answer/positive/negative text
1029            max_train_samples: Maximum training samples
1030            max_eval_samples: Maximum evaluation samples
1031            resample_balanced: Resample to get normal distribution centered at 0
1032            standardize_scores: Standardize score column instead of undersampling
1033            target_std: Target standard deviation for normal distribution (default: 0.3)
1034        """
1035        dataset_type = str(dataset_type).strip().lower()
1036        if dataset_type not in SUPPORTED_SBERT_DATASET_TYPES:
1037            raise ValueError(
1038                f"Unsupported dataset_type='{dataset_type}'. "
1039                f"Supported values: {', '.join(SUPPORTED_SBERT_DATASET_TYPES)}"
1040            )
1041        self.dataset_type = dataset_type
1042        self.query_prefix = str(query_prefix or "")
1043        self.document_prefix = str(document_prefix or "")
1044        self.dataset_columns = self._resolve_dataset_columns(dataset_type, columns or {})
1045
1046        logger.info(f"Loading dataset: {dataset_name}")
1047        logger.info(
1048            "SBERT dataset configuration: type=%s, columns=%s, query_prefix=%r, document_prefix=%r",
1049            self.dataset_type,
1050            self.dataset_columns,
1051            self.query_prefix,
1052            self.document_prefix,
1053        )
1054        
1055        # Load dataset
1056        dataset = load_dataset(dataset_name, split="train")
1057        
1058        logger.info(f"Dataset loaded: {len(dataset)} total samples")
1059        self._validate_required_columns(dataset)
1060        
1061        # Resample to balance distribution (avoid imbalance as recommended)
1062        if (resample_balanced or standardize_scores) and self.dataset_type != "paired_similarity":
1063            logger.warning(
1064                "Score balancing options are only supported for paired_similarity datasets; disabling for dataset_type='%s'",

1138    def _resample_balanced(
1139        self,
1140        dataset,
1141        score_column: str,
1142        bins: int = 20,
1143        target_std: float = 0.3,
1144    ):
1145        """
1146        Resample dataset to create a normal distribution centered at 0.
1147        
1148        This undersamples to avoid imbalanced data as recommended by the dataset authors.
1149        The final distribution will approximate N(0, target_std) in the [-1, 1] range.
1150        
1151        Args:
1152            dataset: The dataset to resample
1153            bins: Number of bins for score distribution (more bins = finer control)
1154            target_std: Target standard deviation for the normal distribution
1155        """
1156        from scipy import stats
1157        
1158        scores = np.array(dataset[score_column], dtype=np.float32)
1159        
1160        logger.info(f"Original distribution: mean={scores.mean():.4f}, std={scores.std():.4f}")
1161        logger.info(f"Score range: [{scores.min():.4f}, {scores.max():.4f}]")
1162        
1163        # Create bins
1164        hist, bin_edges = np.histogram(scores, bins=bins)
1165        # Target: Normal distribution centered at 0 with specified std
1166        # Calculate expected count per bin based on normal distribution
1167        norm_dist = stats.norm(loc=0, scale=target_std)
1168        
1169        # Get probability for each bin
1170        bin_probs = []
1171        for i in range(bins):
1172            prob = norm_dist.cdf(bin_edges[i + 1]) - norm_dist.cdf(bin_edges[i])
1173            bin_probs.append(prob)
1174        
1175        bin_probs = np.array(bin_probs)
1176        bin_probs = bin_probs / bin_probs.sum()  # Normalize
1177        
1178        # Calculate target samples per bin
1179        # Use undersampling: limit by the most restrictive bin
1180        samples_if_all_filled = hist / bin_probs
1181        samples_if_all_filled[bin_probs < 1e-6] = np.inf  # Ignore near-zero probability bins
1182        
1183        # Total samples is limited by the bin with lowest samples/probability ratio
1184        total_target = int(np.min(samples_if_all_filled[np.isfinite(samples_if_all_filled)]))
1185        
1186        # Now calculate per-bin targets
1187        target_per_bin = (bin_probs * total_target).astype(int)
1188        
1189        logger.info(f"Undersampling from {len(scores)} to ~{total_target} samples")
1190        
1191        indices_to_keep = []
1192        for i in range(bins):
1193            # Get indices in this bin
1194            if i < bins - 1:
1195                in_bin = np.where((scores >= bin_edges[i]) & (scores < bin_edges[i + 1]))[0]
1196            else:
1197                in_bin = np.where((scores >= bin_edges[i]) & (scores <= bin_edges[i + 1]))[0]
1198            
1199            target = target_per_bin[i]
1200            
1201            # Sample from bin (undersample if needed)
1202            if len(in_bin) > target and target > 0:
1203                sampled = np.random.choice(in_bin, target, replace=False)
1204            elif target > 0:
1205                sampled = in_bin
1206            else:
1207                continue
1208            
1209            indices_to_keep.extend(sampled.tolist())
1210        
1211        resampled = dataset.select(indices_to_keep)
1212        resampled_scores = np.array(
1213            [float(resampled[i][score_column]) for i in range(len(resampled))],
1214            dtype=np.float32,
1215        )
1216        
1217        logger.info(f"Resampled distribution: mean={resampled_scores.mean():.4f}, std={resampled_scores.std():.4f}")
1218        logger.info(f"Final sample count: {len(resampled)}")
1219        
1220        return resampled

1222    def _standardize_scores(
1223        self,
1224        dataset,
1225        score_column: str,
1226        target_std: float = 0.3,
1227    ):
1228        """
1229        Standardize score magnitudes while preserving original sign and keeping all rows.
1230        """
1231        scores = np.array(dataset[score_column], dtype=np.float32)
1232        source_mean = float(scores.mean())
1233        source_std = float(scores.std())
1234        logger.info(
1235            "Standardizing scores from mean=%.4f, std=%.4f to target std=%.4f with sign preservation",
1236            source_mean,
1237            source_std,
1238            float(target_std),
1239        )
1240        if source_std < 1e-8:
1241            logger.warning(
1242                "Score standard deviation is near zero (%.8f); skipping standardization.",
1243                source_std,
1244            )
1245            return dataset
1246
1247        eps = 1e-12
1248
1249        def _map_batch(batch):
1250            values = np.array(batch[score_column], dtype=np.float32)
1251            standardized = ((values - source_mean) / (source_std + eps)) * float(target_std)
1252            # Preserve original sign after standardization.
1253            signed = np.where(values == 0.0, 0.0, np.sign(values) * np.abs(standardized))
1254            clipped = np.clip(signed, -1.0, 1.0)
1255            return {score_column: clipped.tolist()}
1256
1257        standardized = dataset.map(
1258            _map_batch,
1259            batched=True,
1260            desc=f"Standardizing '{score_column}'",
1261        )
1262        standardized_scores = np.array(standardized[score_column], dtype=np.float32)
1263        logger.info(
1264            "Standardized distribution: mean=%.4f, std=%.4f, range=[%.4f, %.4f]",
1265            float(standardized_scores.mean()),
1266            float(standardized_scores.std()),
1267            float(standardized_scores.min()),
1268            float(standardized_scores.max()),
1269        )
1270        return standardized

1528            fit_kwargs["checkpoint_save_total_limit"] = self.checkpoint_save_total_limit

1537            fit_params = fit_signature.parameters

1540                for param in fit_params.values()

1544            accepts_var_kwargs = False

1573            checkpoint_root = fit_kwargs_cpu.get("checkpoint_path")
1574            if not checkpoint_root:
1575                checkpoint_root = os.path.join(self.output_dir, "checkpoints")
1576                fit_kwargs_cpu["checkpoint_path"] = checkpoint_root
1577
1578            latest = self._find_latest_rolling_checkpoint()
1579            if latest is not None:
1580                fit_kwargs_cpu["resume_from_checkpoint"] = str(latest["path"])
1581            else:
1582                fit_kwargs_cpu.pop("resume_from_checkpoint", None)
1583            self.model.fit(**fit_kwargs_cpu)
1584        
1585        logger.info(f"Training completed! Model saved to {self.output_dir}")
1586        
1587        final_score = None
1588        if self.evaluator is not None:
1589            # Final evaluation
1590            logger.info("Running final evaluation...")
1591            final_score = self.evaluator(self.model)
1592            logger.info("Final Spearman correlation: %s", _format_score_for_log(final_score))
1593        else:
1594            logger.info("Skipping final evaluator for dataset type '%s'", self.dataset_type)
1595        
1596        self.training_metadata['end_time'] = datetime.now().isoformat()
1597        self.training_metadata['final_score'] = final_score
1598        self.training_metadata['final_score_value'] = _extract_primary_score(final_score)
1599        
1600        return final_score

Trainer for SBERT fine-tuning with cosine similarity loss.
